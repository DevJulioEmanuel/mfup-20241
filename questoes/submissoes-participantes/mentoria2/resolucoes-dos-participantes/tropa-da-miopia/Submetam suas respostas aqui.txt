QUESTAO 06 - FIBONACCI:

#include <stdio.h>
 
int main() {
 
    int n;
    scanf("%d", &n);
    
    for (int i=0; i<n; i++) {
        int num;
        scanf("%d", &num);
        long long int vetor[num];
        for (int j=0; j<num; j++) {
            if (j==0) {
                vetor[j]=1;
                continue;
            }
            if (j==1) {
                vetor[j]=1;
                continue;
            }
            if (j>1) {
                vetor[j] = vetor[j-2] + vetor[j-1];
            }
        }
        if (num>0) printf("Fib(%d) = %lld\n", num, vetor[num-1]);
        else printf("Fib(0) = 0\n");
        
    }
 
    return 0;
}
